🚀 Configuración Inicial

Este script preparará las tablas necesarias en tu base de datos de Neon para que el login y los resultados funcionen correctamente.

Tablas a crear:

-- 1. Tabla de Usuarios
CREATE TABLE IF NOT EXISTS usuarios (
    id SERIAL PRIMARY KEY,
    nombre VARCHAR(150),
    tipo_doc VARCHAR(5),
    num_doc VARCHAR(20) UNIQUE,
    cargo VARCHAR(30),
    area VARCHAR(10),
    fecha TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 2. Tabla de Resultados Generales
CREATE TABLE IF NOT EXISTS resultados_evaluacion (
    id SERIAL PRIMARY KEY,
    usuario_id INTEGER REFERENCES usuarios(id),
    porcentaje_total DECIMAL,
    modulo_1 DECIMAL,
    modulo_2 DECIMAL,
    modulo_3 DECIMAL,
    modulo_4 DECIMAL,
    modulo_5 DECIMAL,
    modulo_6 DECIMAL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 3. Tabla de Resultados por Módulo
CREATE TABLE IF NOT EXISTS resultados_modulo_especifico (
    id SERIAL PRIMARY KEY,
    usuario_id INTEGER REFERENCES usuarios(id),
    modulo_nombre VARCHAR(100),
    porcentaje_total DECIMAL,
    total_correctas INTEGER,
    total_preguntas INTEGER,
    respuestas_usuario JSONB,
    respuestas_correctas JSONB,
    detalle_resultado TEXT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
